void setup() {
  // put your setup code here, to run once:
  pinMode(12, OUTPUT);
  
}

void loop() {
  // put your main code here, to run repeatedly:
  for (int i = 0; i<=30; i++){
    do{
    digitalWrite(13,1);
    delay(150);
    }
    while (i%=2);
    i++;
    if (i>=30){
      i = 0;
      }
  }
  for (int j = 0; j<=15; j++){
    digitalWrite(13,0);
    delay(800);
    i++;
  }
}
